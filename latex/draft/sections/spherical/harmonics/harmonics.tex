For the two-sphere, the spherical harmonics%
\footnote{%
  \cite{cheng1999fast} \textsc{Cheng}, \textsc{Greengard}, \& \textsc{Rohlin}  (1999), eq.4
}

\begin{subequations}\label{eq:sphericalharmonics}
  \begin{equation}
    \spherical_{n}^{m}(\alpha^{1}, \alpha^{2}) = c \legendre_{n}^{\absl{m}}{(\cos{\alpha^1})} \exp{(im\alpha^2)}, \qquad c = \sqrt{\frac{(n - \absl{m})!}{(n + \absl{m})!}}
  \end{equation}
\end{subequations}

\noindent
constitute a complete orthonormal basis when $n \geq m$.%
